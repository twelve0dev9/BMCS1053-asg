SET LINESIZE 120
SET PAGESIZE 35

TTITLE CENTER 'Employee Details with Job Position and Organization' SKIP 2

COLUMN EmployeeID HEADING 'Employee ID' FORMAT A10
COLUMN EmpName HEADING 'Employee Name' FORMAT A25
COLUMN JobPosition HEADING 'Job Position' FORMAT A25
COLUMN Organization HEADING 'Organization' FORMAT A30

PROMPT Please enter an Employee ID (e.g., E0001)
ACCEPT emp_id CHAR PROMPT 'Enter Employee ID: '

SELECT 
    e.EmployeeID, 
    e.Name, 
    jp.Position AS JobPosition, 
    o.Org_Name AS Organization
FROM Employees e
JOIN Job_Positions jp ON e.JobPost_ID = jp.JobPost_ID
JOIN Organizations o ON e.OrgID = o.OrgID
WHERE TRIM(e.EmployeeID) = UPPER('&emp_id');

TTITLE OFF;
