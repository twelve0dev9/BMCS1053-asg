SELECT 
    e.employeeID, 
    e.name,
    e.age,
    e.gender,
    e.PhoneNumber,
    e.email,
    j.position, 
    (SELECT i.title
    FROM Industries i
    JOIN Job_Positions j
    on j.Industry_ID = i.Industry_ID
    WHERE j.Industry_ID = i.Industry_ID) AS Industry
from employees e
join job_positions j
on e.jobpost_id = j.jobpost_id
