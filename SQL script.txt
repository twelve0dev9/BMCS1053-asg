CREATE TABLE Industries (
    IndustryID NUMBER PRIMARY KEY,
    Title VARCHAR2(20) NOT NULL,
    Description VARCHAR2(50)
);

CREATE TABLE Organizations (
    OrgID CHAR(6) PRIMARY KEY,
    OrgType VARCHAR2(20) NOT NULL CHECK (OrgType IN ('Companies', 'Marketing Firm', 'Universities', 'Consultancy agent', 'F&B provider')),
    OrgName VARCHAR2(35) NOT NULL,
    Email VARCHAR2(25) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    ContactNumber VARCHAR2(15) CHECK (REGEXP_LIKE(ContactNumber, '^60[0-9]+$')),
    IndustryID NUMBER,
    Roles VARCHAR2(25) CHECK (Roles IN ('Exhibitor', 'Sponsor', 'Speech provider', 'Consultancy provider')),
    nTickets NUMBER,
    CONSTRAINT fk_org_industry FOREIGN KEY (IndustryID) REFERENCES Industries(IndustryID),
    CONSTRAINT chk_org_not_null CHECK (OrgType IS NOT NULL AND OrgName IS NOT NULL)
);

CREATE TABLE Location (
    LocationID NUMBER PRIMARY KEY,
    OrgID CHAR(6),
    LocationName VARCHAR2(25) NOT NULL,
    Floor NUMBER,
    Section NUMBER,
    LocationType VARCHAR2(20) NOT NULL CHECK (LocationType IN ('Main Exhibition Halls', 'Auditoriums', 'Meeting Rooms', 'Networking Lounge', 'Conference Rooms')),
    RentalPrice NUMBER,
    "SIZE" VARCHAR2(6) CHECK ("SIZE" IN ('Small', 'Medium', 'Large')),
    CONSTRAINT fk_location_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_location_not_null CHECK (LocationID IS NOT NULL AND LocationName IS NOT NULL AND LocationType IS NOT NULL)
);

CREATE TABLE Events (
    EventID NUMBER PRIMARY KEY,
    Title VARCHAR2(35) NOT NULL,
    Description VARCHAR2(50),
    EventType VARCHAR2(30) NOT NULL CHECK (EventType IN ('Recruitment', 'Networking Sessions', 'Talks', 'Seminars', 'Career Consultation')),
    EventTime DATE,
    LocationID NUMBER,
    CONSTRAINT fk_event_location FOREIGN KEY (LocationID) REFERENCES Location(LocationID),
    CONSTRAINT chk_event_not_null CHECK (EventID IS NOT NULL AND Title IS NOT NULL AND EventType IS NOT NULL)
);

CREATE TABLE Speakers (
    SpeakerID NUMBER PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    PhoneNumber VARCHAR2(15) CHECK (REGEXP_LIKE(PhoneNumber, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    Affiliation VARCHAR2(25),
    EventID NUMBER NOT NULL,
    CONSTRAINT fk_speaker_event FOREIGN KEY (EventID) REFERENCES Events(EventID),
    CONSTRAINT chk_speaker_not_null CHECK (EventID IS NOT NULL)
);

CREATE TABLE Applicants (
    ApplicantID CHAR(5) PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    PhoneNumber VARCHAR2(15) CHECK (REGEXP_LIKE(PhoneNumber, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    EducationLevel VARCHAR2(10),
    Major VARCHAR2(25),
    University CHAR(6),
    "Resume" VARCHAR2(20),
    Portfolio VARCHAR2(20),
    CONSTRAINT fk_applicant_university FOREIGN KEY (University) REFERENCES Organizations(OrgID)
);

CREATE TABLE JobPositions (
    JobPostID NUMBER PRIMARY KEY,
    IndustryID NUMBER NOT NULL,
    Position CHAR(25) NOT NULL,
    CONSTRAINT fk_jobpos_industry FOREIGN KEY (IndustryID) REFERENCES Industries(IndustryID),
    CONSTRAINT chk_position_values CHECK (Position IN ('Speaker', 'Panelist', 'Recruiter', 'Interviewer', 'Consultant', 'General staff', 'Planner', 'Coordinator', 'Logistics', 'Partnership management', 'F&B manager', 'F&B general staff'))
);

CREATE TABLE Employees (
    EmployeeID CHAR(5) PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    PhoneNumber VARCHAR2(15) CHECK (REGEXP_LIKE(PhoneNumber, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    OrgID CHAR(6) NOT NULL,
    JobPostID NUMBER NOT NULL,
    CONSTRAINT fk_employee_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT fk_employee_jobpost FOREIGN KEY (JobPostID) REFERENCES JobPositions(JobPostID)
);

CREATE TABLE Applications (
    OrgID CHAR(6) NOT NULL,
    ApplicantID CHAR(5) NOT NULL,
    JobPostID NUMBER NOT NULL,
    EmploymentTypes VARCHAR2(15) CHECK (EmploymentTypes IN ('Internships', 'Part time', 'Full time')),
    ApplicationStatus VARCHAR2(20) DEFAULT 'Applied' CHECK (ApplicationStatus IN ('Applied', 'Resume Submitted', 'Interview scheduled', 'Rejected', 'Interview done')),
    DateApplied DATE,
    CONSTRAINT pk_applications PRIMARY KEY (OrgID, ApplicantID, JobPostID),
    CONSTRAINT fk_application_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT fk_applicants FOREIGN KEY (ApplicantID) REFERENCES Applicants(ApplicantID),
    CONSTRAINT fk_application_jobpost FOREIGN KEY (JobPostID) REFERENCES JobPositions(JobPostID),
    CONSTRAINT chk_application_not_null CHECK (OrgID IS NOT NULL AND ApplicantID IS NOT NULL AND JobPostID IS NOT NULL)
);

CREATE TABLE EventsParticipation (
    EventID NUMBER PRIMARY KEY, 
    SpeakerID NUMBER, 
    OrgID CHAR(6), 
    CONSTRAINT fk_Event FOREIGN KEY (EventID) REFERENCES Events(EventID), 
    CONSTRAINT fk_Speaker FOREIGN KEY (SpeakerID) REFERENCES Speakers(SpeakerID), 
    CONSTRAINT fk_Org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID)
);

CREATE TABLE Participations (
    PartID NUMBER PRIMARY KEY,
    SpeakerID NUMBER NOT NULL,
    OrgID CHAR(6) NOT NULL,
    ParticipateBy VARCHAR2(8) CHECK (ParticipateBy IN ('Applying', 'Invited')),
    DateIA DATE,
    DateConfirmed DATE,
    Status VARCHAR2(10) CHECK (Status IN ('Invited', 'Applied', 'Declined', 'Accepted')),
    CONSTRAINT fk_participation_speaker FOREIGN KEY (SpeakerID) REFERENCES Speakers(SpeakerID),
    CONSTRAINT fk_participation_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_participation_not_null CHECK (PartID IS NOT NULL AND OrgID IS NOT NULL)
);

CREATE TABLE AdCampaign (
    CampaignID CHAR(4) PRIMARY KEY,
    OrgID CHAR(6) NOT NULL,
    Title VARCHAR2(25) NOT NULL,
    StartDate DATE,
    EndDate DATE,
    Channel VARCHAR2(40) CHECK (Channel IN ('Social media marketing', 'Outdoor advertising', 'Broadcast ads', 'Event branding on e-hailing vehicles')),
    Budget NUMBER,
    TargetAud VARCHAR2(20),
    CONSTRAINT fk_adcampaign_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_adcampaign_not_null CHECK (CampaignID IS NOT NULL AND OrgID IS NOT NULL AND Title IS NOT NULL)
);
