CREATE TABLE Industries (
    Industry_ID NUMBER PRIMARY KEY,
    Title VARCHAR2(20) NOT NULL,
    Description VARCHAR2(50)
);

CREATE TABLE Organizations (
    OrgID CHAR(6) PRIMARY KEY,
    Org_Name VARCHAR2(35) NOT NULL,
    Email VARCHAR2(45) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    Contact_Number VARCHAR2(15) CHECK (REGEXP_LIKE(Contact_Number, '^60[0-9]+$')),
    Industry_ID NUMBER,
    Roles VARCHAR2(25) CHECK (Roles IN ('Exhibitor', 'Sponsor', 'Speech provider', 'Consultancy provider')),
    nTickets NUMBER,
    CONSTRAINT fk_org_industry FOREIGN KEY (Industry_ID) REFERENCES Industries(Industry_ID),
    CONSTRAINT chk_org_not_null CHECK (Org_Name IS NOT NULL)
);

CREATE TABLE Location (
    Location_ID NUMBER PRIMARY KEY,
    OrgID CHAR(6),
    Location_Name VARCHAR2(25) NOT NULL,
    Floor NUMBER,
    Section NUMBER,
    Location_Type VARCHAR2(30) NOT NULL CHECK (Location_Type IN ('Main Exhibition Halls', 'Auditoriums', 'Meeting Rooms', 'Networking Lounge', 'Conference Rooms')),
    Rental_Price NUMBER,
    "SIZE" VARCHAR2(6) CHECK ("SIZE" IN ('Small', 'Medium', 'Large')),
    CONSTRAINT fk_location_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_location_not_null CHECK (Location_ID IS NOT NULL AND Location_Name IS NOT NULL AND Location_Type IS NOT NULL)
);

CREATE TABLE Events ( 
    Event_ID NUMBER PRIMARY KEY,
    Title VARCHAR2(35) NOT NULL,
    Description VARCHAR2(70),
    Event_Type VARCHAR2(30) NOT NULL,
    Event_Time DATE,
    Location_ID NUMBER,
    CONSTRAINT fk_event_location FOREIGN KEY (Location_ID) REFERENCES Location(Location_ID),
    CONSTRAINT chk_event_type CHECK (Event_Type IN ('Recruitment', 'Networking Sessions', 'Talks', 'Seminars', 'Career Consultation', 'Panels')), 
    CONSTRAINT chk_event_not_null CHECK (Event_ID IS NOT NULL AND Title IS NOT NULL AND Event_Type IS NOT NULL)
);

CREATE TABLE Speakers (
    Speaker_ID NUMBER PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    Phone_Number VARCHAR2(15) CHECK (REGEXP_LIKE(Phone_Number, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    Affiliation VARCHAR2(25),
    Event_ID NUMBER NOT NULL,
    CONSTRAINT fk_speaker_event FOREIGN KEY (Event_ID) REFERENCES Events(Event_ID),
    CONSTRAINT chk_speaker_not_null CHECK (Event_ID IS NOT NULL)
);

CREATE TABLE Applicants (
    Applicant_ID CHAR(5) PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    Phone_Number VARCHAR2(15) CHECK (REGEXP_LIKE(Phone_Number, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    Education_Level VARCHAR2(10),
    Major VARCHAR2(25),
    University CHAR(6),
    "Resume" VARCHAR2(20),
    Portfolio VARCHAR2(20),
    CONSTRAINT fk_applicant_university FOREIGN KEY (University) REFERENCES Organizations(OrgID)
);

CREATE TABLE Job_Positions (
    JobPost_ID NUMBER PRIMARY KEY,
    Industry_ID NUMBER,
    Position CHAR(25) NOT NULL,
    CONSTRAINT fk_jobpos_industry FOREIGN KEY (Industry_ID) REFERENCES Industries(Industry_ID)
);

CREATE TABLE Employees (
    EmployeeID CHAR(5) PRIMARY KEY,
    Name VARCHAR2(20) NOT NULL,
    Age NUMBER CHECK (Age >= 0),
    Gender CHAR(1) CHECK (UPPER(Gender) IN ('M', 'F')),
    PhoneNumber VARCHAR2(15) CHECK (REGEXP_LIKE(PhoneNumber, '^60[0-9]+$')),
    Email VARCHAR2(30) CHECK (REGEXP_LIKE(Email, '^[a-zA-Z]\w+@(\S+)$')),
    OrgID CHAR(6) NOT NULL,
    JobPost_ID NUMBER NOT NULL,
    CONSTRAINT fk_employee_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT fk_employee_jobpost FOREIGN KEY (JobPost_ID) REFERENCES Job_Positions(JobPost_ID)
);

CREATE TABLE Applications (
    OrgID CHAR(6) NOT NULL,
    Applicant_ID CHAR(5) NOT NULL,
    JobPost_ID NUMBER NOT NULL,
    Employment_Types VARCHAR2(15) CHECK (Employment_Types IN ('Internships', 'Part time', 'Full time')),
    Application_Status VARCHAR2(20) DEFAULT 'Applied' CHECK (Application_Status IN ('Applied', 'Resume Submitted', 'Interview scheduled', 'Rejected', 'Interview done')),
    DateApplied DATE,
    CONSTRAINT pk_applications PRIMARY KEY (OrgID, Applicant_ID, JobPost_ID),
    CONSTRAINT fk_application_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT fk_applicants FOREIGN KEY (Applicant_ID) REFERENCES Applicants(Applicant_ID),
    CONSTRAINT fk_application_jobpost FOREIGN KEY (JobPost_ID) REFERENCES Job_Positions(JobPost_ID),
    CONSTRAINT chk_application_not_null CHECK (OrgID IS NOT NULL AND Applicant_ID IS NOT NULL AND JobPost_ID IS NOT NULL)
);

CREATE TABLE Events_Participation (
    Event_ID NUMBER, 
    Speaker_ID NUMBER, 
    OrgID CHAR(6), 
    CONSTRAINT fk_Event FOREIGN KEY (Event_ID) REFERENCES Events(Event_ID), 
    CONSTRAINT fk_Speaker FOREIGN KEY (Speaker_ID) REFERENCES Speakers(Speaker_ID), 
    CONSTRAINT fk_Org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT pk_events_participation PRIMARY KEY (Event_ID, Speaker_ID, OrgID)
);

CREATE TABLE Participations (
    PartID NUMBER PRIMARY KEY,
    Speaker_ID NUMBER,
    OrgID CHAR(6) NOT NULL,
    ParticipateBy VARCHAR2(8) CHECK (ParticipateBy IN ('Applying', 'Invited')),
    Date_IA DATE,
    Date_Confirmed DATE,
    Status VARCHAR2(10) CHECK (Status IN ('Invited', 'Applied', 'Declined', 'Accepted')),
    CONSTRAINT fk_participation_speaker FOREIGN KEY (Speaker_ID) REFERENCES Speakers(Speaker_ID),
    CONSTRAINT fk_participation_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_participation_not_null CHECK (PartID IS NOT NULL AND OrgID IS NOT NULL)
);

CREATE TABLE AdCampaign (
    CampaignID CHAR(4) PRIMARY KEY,
    OrgID CHAR(6) NOT NULL,
    Title VARCHAR2(25) NOT NULL,
    StartDate DATE,
    EndDate DATE,
    Channel VARCHAR2(40) CHECK (Channel IN ('Social media marketing', 'Outdoor advertising', 'Broadcast ads', 'Event branding on e-hailing vehicles')),
    Budget NUMBER,
    TargetAud VARCHAR2(20),
    CONSTRAINT fk_adcampaign_org FOREIGN KEY (OrgID) REFERENCES Organizations(OrgID),
    CONSTRAINT chk_adcampaign_not_null CHECK (CampaignID IS NOT NULL AND OrgID IS NOT NULL AND Title IS NOT NULL)
);
