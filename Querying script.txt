-- Format columns for readability
COLUMN "Applicant Name" FORMAT A25 HEADING "Applicant Name"
COLUMN "Major" FORMAT A30 HEADING "Major"
COLUMN "Education Level" FORMAT A20 HEADING "Education Level"
COLUMN "Email" FORMAT A30 HEADING "Email Contact"

-- The main SELECT statement
SELECT
    app.Name AS "Applicant Name",
    app.Major,
    app.Education_Level AS "Education Level",
    app.Email
FROM
    Applicants app
JOIN
    Organizations o ON app.University = o.OrgID
WHERE
    o.Org_Name = '&v_university_name'
ORDER BY
    app.Name;
/

COLUMN "Contact Email" FORMAT A35 HEADING "Contact Email"
COLUMN "Contact Number" FORMAT A20 HEADING "Contact Number"

-- The main SELECT statement with a subquery
SELECT
    Org_Name AS "Marketing Firm Name",
    Email AS "Contact Email",
    Contact_Number AS "Contact Number"
FROM
    Organizations
WHERE
    OrgID IN (SELECT OrgID
              FROM AdCampaign
              WHERE Channel = '&v_channel_type')
ORDER BY
    Org_Name;
