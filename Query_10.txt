-- Define column formatting
COLUMN Speaker_Name FORMAT A25 HEADING 'Speaker Name'
COLUMN Total_Events FORMAT 9999 HEADING 'Total Events'

-- Query to show the number of events per selected speaker
SELECT s.Name AS Speaker_Name,
       COUNT(e.Event_ID) AS Total_Events
FROM Speakers s
JOIN Events_Participation ep ON s.Speaker_ID = ep.Speaker_ID
JOIN Events e ON ep.Event_ID = e.Event_ID
WHERE s.Speaker_ID = &CHOICE
GROUP BY s.Name;
